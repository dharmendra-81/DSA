void nStarDiamond(int n) {
    for (int i = 0; i < 2*n; i++)
    {
        for (int j=0;j<2*n-1;j++)
        {
            if ((n-i-j>1) || (i-j>n)){
              cout<<" ";
            }
            else if((j-i==n) || (i+j==3*n-1)){
              break;
            }
            else{
            cout<<"*";
            }
        }
        cout<<endl;
    }
  }
