class Solution {
    public:
      void printTriangle(int n) {
        for (int i = 0; i < n; i++)
        {
            for (int j=0;j<2*n-1;j++)
            {
                if (n-i-j>1){
                  cout<<" ";
                }
                else if(j-i==n){
                  break;
                }
                else
                cout<<"*";
            }
            cout<<endl;
        }

      }
  };
