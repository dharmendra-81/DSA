Given an unsorted array of integers `nums`, return _the length of the longest consecutive elements sequence_

---
#### Better Approach
```cpp
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        if(nums.empty()) return 0; //meaning no elements in the array
        sort(nums.begin(), nums.end()); //time complexity O(nlogn)
        int lastSmaller = INT_MIN;
        int longest = 1;
        int count = 1;
        for(int i = 0; i < nums.size(); i++){
            if(nums[i] == lastSmaller + 1){
                count++;
                lastSmaller = nums[i];
            } else if(nums[i] != lastSmaller) {
                count = 1;
                lastSmaller = nums[i];
            }
            longest = max(longest, count);
        }
        return longest; 
    }
};
```

#### Optimal O(3N)
```cpp
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        if(nums.size() == 0) return 0;
        int maxLength = 1;
        unordered_set<int> Set;
        for(int i = 0; i < nums.size(); i++){ // O(n)
            Set.insert(nums[i]);
        }
        for(auto it: Set){ 
            if(Set.find(it - 1) == Set.end()){
                int currentNum = it;
                int currentLength = 1;
                while(Set.find(currentNum + 1) != Set.end()){
                    currentNum++;
                    currentLength++;
                }
                maxLength = max(maxLength, currentLength);
            }
        }
        return maxLength;
    }
};
```
