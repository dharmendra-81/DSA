Given an integer array of size `n`, find all elements that appear more than `⌊ n/3 ⌋` times.

###### Better
```cpp
class Solution {
public:
    vector<int> majorityElement(vector<int>& nums) {
       vector<int>ls;
       map<int,int>mpp;
       int n= nums.size();
       int min = (int)(n/3)+1;
       for(int i=0; i<n; i++){
        mpp[nums[i]]++;
        if(mpp[nums[i]] == min){
            ls.push_back(nums[i]);
        }
        if(ls.size() == 2) break;
       } 
       sort(ls.begin(), ls.end());
       return ls;
    }
};
```

###### Optimal
o(2n)
```cpp
class Solution {
public:
    vector<int> majorityElement(vector<int>& nums) {
      int cnt1=0, cnt2=0;
      int el1 = INT_MIN;
      int el2 = INT_MIN;
      int n= nums.size();
      int min = (int)(n/3)+1;
      for(int i=0; i<n; i++){
          if(cnt1 == 0 && el2 != nums[i]){
              cnt1=1;
            el1=nums[i];
        }
        else if(cnt2 == 0 && el1 != nums[i]){
            cnt2=1;
            el2=nums[i];
        }
        else if (nums[i] == el1) cnt1++;
        else if (nums[i] == el2) cnt2++;
        else{
            cnt1--;
            cnt2--;
        }
    }
    vector<int> ls;
    cnt1 = 0, cnt2 = 0;
    for (int i = 0; i < n; i++){
        if (nums[i] == el1)
            cnt1++;
        if (nums[i] == el2)
            cnt2++;
    }
    if (cnt1 >= min)
        ls.push_back(el1);
    if (cnt2 >= min)
        ls.push_back(el2);
    return ls;
    }
};
```
