Given an integer array `nums`, find the sub array with the largest sum, and return _its sum_.

---
```cpp
#include<climits>
class Solution {
    public:
    int maxSubArray(vector<int>& nums) {
        //kadane's algorithm
        int maxSum = INT_MIN;
        int currentSum = 0;   
        for(int i = 0; i < nums.size(); i++) {
            currentSum += nums[i];
            if(currentSum > maxSum) {
                maxSum = currentSum;
            }
            if(currentSum < 0) {
                currentSum = 0;
            }
        }
        return maxSum;
    }
};
```

class Solution {
    public:
    int maxSubArray(vector<int>& nums) {
        int maxSum = INT_MIN;
        int currentSum = 0;  
        int start=0;
        int startIndex = -1, endIndex = -1; 
        for(int i = 0; i < nums.size(); i++) {
            if(currentSum <= 0) {
                start = i; // Reset start index when current sum is non-positive
            }
            currentSum += nums[i];
            if(currentSum > maxSum) {
                maxSum = currentSum;
                startIndex = start; // Update start index when a new max is found
                endIndex = i; // Update end index to current index
            }
            if(currentSum < 0) {
                currentSum = 0;
            }
        }
        cout << "Maximum subarray found from index " << startIndex << " to " << endIndex <<" "<< endl;
        printVector(vector<int>(nums.begin() + startIndex, nums.begin() + endIndex + 1));
        return maxSum;
    }
};
